ðŸŒ“ Light/Dark Mode Banner Switching#

To dynamically switch banners when toggling PaperMod’s theme:

A. layouts/index.html overrides homepage content:#

{{ define "main" }}
<main>
  <p>
    <img id="hero-banner"
         src="/banner.png"
         data-dark="/banner.png"
         data-light="/banner-inverted.png"
         alt="alexbakertech banner"
         style="width:100%; border-radius:8px;" />
  </p>
  {{ .Content }}
</main>
{{ end }}

B. layouts/partials/extend_footer.html injects theme-aware JavaScript:#

<script>
  function updateBannerByTheme() {
    const banner = document.getElementById("hero-banner");
    if (!banner) return;
    const theme = localStorage.getItem("pref-theme") || 
      (document.body.classList.contains("dark") ? "dark" : "light");
    banner.src = (theme === "dark") ? banner.dataset.dark : banner.dataset.light;
  }

  document.getElementById("theme-toggle")?.addEventListener("click", () => {
    setTimeout(updateBannerByTheme, 10);
  });

  updateBannerByTheme();
</script>
